017package org.apache.activemq.transport.amqp.sasl;
018
019/**
020 * Base class for SASL Mechanisms that provides common functionality.
021 */
022public abstract class AbstractSaslMechanism implements SaslMechanism {
023
024    protected String username;
025    protected String password;
026    protected boolean failed;
027    protected String failureReason;
028
029    @Override
030    public String getUsername() {
031        return username;
032    }
033
034    @Override
035    public String getPassword() {
036        return password;
037    }
038
039    @Override
040    public boolean isFailed() {
041        return failed;
042    }
043
044    @Override
045    public String getFailureReason() {
046        return failureReason;
047    }
048
049    void setFailed(String failureReason) {
050        this.failed = true;
051        this.failureReason = failureReason;
052    }
053}
