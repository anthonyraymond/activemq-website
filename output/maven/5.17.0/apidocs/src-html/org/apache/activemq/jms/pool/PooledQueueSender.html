017package org.apache.activemq.jms.pool;
018
019import javax.jms.Destination;
020import javax.jms.JMSException;
021import javax.jms.Message;
022import javax.jms.Queue;
023import javax.jms.QueueSender;
024
025/**
026 * {@link QueueSender} instance that is created and managed by the PooledConnection.
027 */
028public class PooledQueueSender extends PooledProducer implements QueueSender {
029
030    public PooledQueueSender(QueueSender messageProducer, Destination destination) throws JMSException {
031        super(messageProducer, destination);
032    }
033
034    @Override
035    public void send(Queue queue, Message message, int i, int i1, long l) throws JMSException {
036        getQueueSender().send(queue, message, i, i1, l);
037    }
038
039    @Override
040    public void send(Queue queue, Message message) throws JMSException {
041        getQueueSender().send(queue, message);
042    }
043
044    @Override
045    public Queue getQueue() throws JMSException {
046        return (Queue) getDestination();
047    }
048
049    protected QueueSender getQueueSender() {
050        return (QueueSender) getMessageProducer();
051    }
052}
