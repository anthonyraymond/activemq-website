017package org.apache.activemq.filter;
018
019import java.lang.IllegalStateException;
020import org.apache.activemq.command.ActiveMQDestination;
021
022/*
023  * allow match to any set  of composite destinations, both queues and topics
024 */
025public class AnyDestination extends ActiveMQDestination {
026
027    public AnyDestination(ActiveMQDestination[] destinations) {
028        super(destinations);
029        // ensure we are small when it comes to comparison in DestinationMap
030        physicalName = "!0";
031    }
032
033    @Override
034    protected String getQualifiedPrefix() {
035        return "Any://";
036    }
037
038    @Override
039    public byte getDestinationType() {
040        return ActiveMQDestination.QUEUE_TYPE & ActiveMQDestination.TOPIC_TYPE;
041    }
042
043    @Override
044    public byte getDataStructureType() {
045        throw new IllegalStateException("not for marshalling");
046    }
047
048    @Override
049    public boolean isQueue() {
050        return true;
051    }
052
053    @Override
054    public boolean isTopic() {
055        return true;
056    }
057}
