017package org.apache.activemq.transport.discovery.zeroconf;
018
019import java.io.IOException;
020import java.net.URI;
021import java.util.Map;
022
023import org.apache.activemq.transport.discovery.DiscoveryAgent;
024import org.apache.activemq.transport.discovery.DiscoveryAgentFactory;
025import org.apache.activemq.util.IOExceptionSupport;
026import org.apache.activemq.util.IntrospectionSupport;
027import org.apache.activemq.util.URISupport;
028
029public class ZeroconfDiscoveryAgentFactory extends DiscoveryAgentFactory {
030
031    protected DiscoveryAgent doCreateDiscoveryAgent(URI uri) throws IOException {
032        try {
033            Map options = URISupport.parseParameters(uri);
034            ZeroconfDiscoveryAgent rc = new ZeroconfDiscoveryAgent();
035            rc.setGroup(uri.getHost());
036            IntrospectionSupport.setProperties(rc, options);
037            return rc;
038            
039        } catch (Throwable e) {
040            throw IOExceptionSupport.create("Could not create discovery agent: " + uri, e);
041        }
042    }
043}
