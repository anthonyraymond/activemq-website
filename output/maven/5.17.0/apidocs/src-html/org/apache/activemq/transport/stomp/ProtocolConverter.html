149    protected ResponseHandler createResponseHandler(final StompFrame command) {
150        final String receiptId = command.getHeaders().get(Stomp.Headers.RECEIPT_REQUESTED);
151        if (receiptId != null) {
152            return new ResponseHandler() {
153                @Override
154                public void onResponse(ProtocolConverter converter, Response response) throws IOException {
155                    if (response.isException()) {
156                        // Generally a command can fail.. but that does not invalidate the connection.
157                        // We report back the failure but we don't close the connection.
158                        Throwable exception = ((ExceptionResponse)response).getException();
159                        handleException(exception, command);
160                    } else {
161                        StompFrame sc = new StompFrame();
162                        sc.setAction(Stomp.Responses.RECEIPT);
163                        sc.setHeaders(new HashMap<String, String>(1));
164                        sc.getHeaders().put(Stomp.Headers.Response.RECEIPT_ID, receiptId);
165                        stompTransport.sendToStomp(sc);
166                    }
167                }
168            };
169        }
170        return null;
171    }

174        command.setCommandId(generateCommandId());

176            command.setResponseRequired(true);
177            resposeHandlers.put(Integer.valueOf(command.getCommandId()), handler);
178        }
179        stompTransport.sendToActiveMQ(command);
180    }

190    protected FrameTranslator findTranslator(String header, ActiveMQDestination destination, boolean advisory) {
191        FrameTranslator translator = frameTranslator;
192        try {
193            if (header != null) {
194               translator=jmsFrameTranslators.get(header);
195                if(translator==null) {
196                        LOG.info("Creating a new FrameTranslator to convert "+header);
197                        translator = (FrameTranslator) FRAME_TRANSLATOR_FINDER.newInstance(header);
198                        if(translator!=null) {
199                                LOG.info("Created a new FrameTranslator to convert "+header);
200                                jmsFrameTranslators.put(header,translator);
201                        }else {
202                                LOG.error("Failed in creating FrameTranslator to convert "+header);
203                        }                               
204                }
205            } else {
206                if (destination != null && (advisory || AdvisorySupport.isAdvisoryTopic(destination))) {
207                    translator = new JmsFrameTranslator();
208                }
209            }
210        } catch (Exception ignore) {
211            // if anything goes wrong use the default translator
212                        LOG.debug("Failed in getting a FrameTranslator to convert ", ignore);
213                translator = frameTranslator;
214        }
215
216        if (translator instanceof BrokerContextAware) {
217            ((BrokerContextAware)translator).setBrokerContext(brokerContext);
218        }
219
220        return translator;
221    }

227     */
228    public void onStompCommand(StompFrame command) throws IOException, JMSException {
229        try {
230
231            if (command.getClass() == StompFrameError.class) {
232                throw ((StompFrameError)command).getException();
233            }
234
235            String action = command.getAction();
236            if (action.startsWith(Stomp.Commands.SEND)) {
237                onStompSend(command);
238            } else if (action.startsWith(Stomp.Commands.ACK)) {
239                onStompAck(command);
240            } else if (action.startsWith(Stomp.Commands.NACK)) {
241                onStompNack(command);
242            } else if (action.startsWith(Stomp.Commands.BEGIN)) {
243                onStompBegin(command);
244            } else if (action.startsWith(Stomp.Commands.COMMIT)) {
245                onStompCommit(command);
246            } else if (action.startsWith(Stomp.Commands.ABORT)) {
247                onStompAbort(command);
248            } else if (action.startsWith(Stomp.Commands.SUBSCRIBE_PREFIX)) {
249                onStompSubscribe(command);
250            } else if (action.startsWith(Stomp.Commands.UNSUBSCRIBE_PREFIX)) {
251                onStompUnsubscribe(command);
252            } else if (action.startsWith(Stomp.Commands.CONNECT) ||
253                       action.startsWith(Stomp.Commands.STOMP)) {
254                onStompConnect(command);
255            } else if (action.startsWith(Stomp.Commands.DISCONNECT)) {
256                onStompDisconnect(command);
257            } else {
258                throw new ProtocolException("Unknown STOMP action: " + action, true);
259            }
260
261        } catch (ProtocolException e) {
262            handleException(e, command);
263            // Some protocol errors can cause the connection to get closed.
264            if (e.isFatal()) {
265               getStompTransport().onException(e);
266            }
267        }
268    }
269
270    protected void handleException(Throwable exception, StompFrame command) throws IOException {
271        if (command == null) {
272            LOG.warn("Exception occurred while processing a command: {}", exception.toString());
273        } else {
274            if (exception instanceof JMSException) {
275                JMSException jmsException = (JMSException) exception;
276                if (jmsException.getLinkedException() != null) {
277                    LOG.warn("Exception occurred for client {} ({}) processing: {} -> {} ({})", connectionInfo.getClientId(), connectionInfo.getClientIp(), safeGetAction(command), exception.toString(), jmsException.getLinkedException().toString());
278                } else {
279                    LOG.warn("Exception occurred for client {} ({}) processing: {} -> {}", connectionInfo.getClientId(), connectionInfo.getClientIp(), safeGetAction(command), exception.toString());
280                }
281            } else {
282                LOG.warn("Exception occurred for client {} ({}) processing: {} -> {}", connectionInfo.getClientId(), connectionInfo.getClientIp(), safeGetAction(command), exception.toString());
283            }
284        }
285
286        if (LOG.isDebugEnabled()) {
287            LOG.debug("Exception detail", exception);
288        }
289
290        if (command != null && LOG.isTraceEnabled()) {
291            LOG.trace("Command that caused the error: {}", command);
292        }
293
294        // Let the stomp client know about any protocol errors.
295        ByteArrayOutputStream baos = new ByteArrayOutputStream();
296        PrintWriter stream = new PrintWriter(new OutputStreamWriter(baos, "UTF-8"));
297        if (exception instanceof SecurityException || exception.getCause() instanceof SecurityException) {
298            stream.write(exception.getLocalizedMessage());
299        } else {
300            exception.printStackTrace(stream);
301        }
302        stream.close();
303
304        HashMap<String, String> headers = new HashMap<>();
305        headers.put(Stomp.Headers.Error.MESSAGE, exception.getMessage());
306        headers.put(Stomp.Headers.CONTENT_TYPE, "text/plain");
307
308        if (command != null) {
309            final String receiptId = command.getHeaders().get(Stomp.Headers.RECEIPT_REQUESTED);
310            if (receiptId != null) {
311                headers.put(Stomp.Headers.Response.RECEIPT_ID, receiptId);
312            }
313        }
314
315        StompFrame errorMessage = new StompFrame(Stomp.Responses.ERROR, headers, baos.toByteArray());
316        sendToStomp(errorMessage);
317    }

319    protected void onStompSend(StompFrame command) throws IOException, JMSException {
320        checkConnected();
321
322        Map<String, String> headers = command.getHeaders();
323        String destination = headers.get(Stomp.Headers.Send.DESTINATION);
324        if (destination == null) {
325            throw new ProtocolException("SEND received without a Destination specified!");
326        }
327
328        String stompTx = headers.get(Stomp.Headers.TRANSACTION);
329        headers.remove("transaction");
330
331        ActiveMQMessage message = convertMessage(command);
332
333        message.setProducerId(producerId);
334        MessageId id = new MessageId(producerId, messageIdGenerator.getNextSequenceId());
335        message.setMessageId(id);
336
337        if (stompTx != null) {
338            TransactionId activemqTx = transactions.get(stompTx);
339            if (activemqTx == null) {
340                throw new ProtocolException("Invalid transaction id: " + stompTx);
341            }
342            message.setTransactionId(activemqTx);
343        }
344
345        message.onSend();
346        message.beforeMarshall(null);
347        sendToActiveMQ(message, createResponseHandler(command));
348    }

350    protected void onStompNack(StompFrame command) throws ProtocolException {
351
352        checkConnected();
353
354        if (this.version.equals(Stomp.V1_0)) {
355            throw new ProtocolException("NACK received but connection is in v1.0 mode.");
356        }
357
358        Map<String, String> headers = command.getHeaders();
359
360        String subscriptionId = headers.get(Stomp.Headers.Ack.SUBSCRIPTION);
361        if (subscriptionId == null && !this.version.equals(Stomp.V1_2)) {
362            throw new ProtocolException("NACK received without a subscription id for acknowledge!");
363        }
364
365        String messageId = headers.get(Stomp.Headers.Ack.MESSAGE_ID);
366        if (messageId == null && !this.version.equals(Stomp.V1_2)) {
367            throw new ProtocolException("NACK received without a message-id to acknowledge!");
368        }
369
370        String ackId = headers.get(Stomp.Headers.Ack.ACK_ID);
371        if (ackId == null && this.version.equals(Stomp.V1_2)) {
372            throw new ProtocolException("NACK received without an ack header to acknowledge!");
373        }
374
375        TransactionId activemqTx = null;
376        String stompTx = headers.get(Stomp.Headers.TRANSACTION);
377        if (stompTx != null) {
378            activemqTx = transactions.get(stompTx);
379            if (activemqTx == null) {
380                throw new ProtocolException("Invalid transaction id: " + stompTx);
381            }
382        }
383
384        boolean nacked = false;
385
386        if (ackId != null) {
387            StompAckEntry pendingAck = this.pendingAcks.get(ackId);
388            if (pendingAck != null) {
389                messageId = pendingAck.getMessageId().toString();
390                MessageAck ack = pendingAck.onMessageNack(activemqTx);
391                if (ack != null) {
392                    sendToActiveMQ(ack, createResponseHandler(command));
393                    nacked = true;
394                }
395            }
396        } else if (subscriptionId != null) {
397            StompSubscription sub = this.subscriptions.get(subscriptionId);
398            if (sub != null) {
399                MessageAck ack = sub.onStompMessageNack(messageId, activemqTx);
400                if (ack != null) {
401                    sendToActiveMQ(ack, createResponseHandler(command));
402                    nacked = true;
403                }
404            }
405        }
406
407        if (!nacked) {
408            throw new ProtocolException("Unexpected NACK received for message-id [" + messageId + "]");
409        }
410    }

412    protected void onStompAck(StompFrame command) throws ProtocolException {
413        checkConnected();
414
415        Map<String, String> headers = command.getHeaders();
416        String messageId = headers.get(Stomp.Headers.Ack.MESSAGE_ID);
417        if (messageId == null && !(this.version.equals(Stomp.V1_2))) {
418            throw new ProtocolException("ACK received without a message-id to acknowledge!");
419        }
420
421        String subscriptionId = headers.get(Stomp.Headers.Ack.SUBSCRIPTION);
422        if (subscriptionId == null && this.version.equals(Stomp.V1_1)) {
423            throw new ProtocolException("ACK received without a subscription id for acknowledge!");
424        }
425
426        String ackId = headers.get(Stomp.Headers.Ack.ACK_ID);
427        if (ackId == null && this.version.equals(Stomp.V1_2)) {
428            throw new ProtocolException("ACK received without a ack id for acknowledge!");
429        }
430
431        TransactionId activemqTx = null;
432        String stompTx = headers.get(Stomp.Headers.TRANSACTION);
433        if (stompTx != null) {
434            activemqTx = transactions.get(stompTx);
435            if (activemqTx == null) {
436                throw new ProtocolException("Invalid transaction id: " + stompTx);
437            }
438        }
439
440        boolean acked = false;
441
442        if (ackId != null) {
443            StompAckEntry pendingAck = this.pendingAcks.get(ackId);
444            if (pendingAck != null) {
445                messageId = pendingAck.getMessageId().toString();
446                MessageAck ack = pendingAck.onMessageAck(activemqTx);
447                if (ack != null) {
448                    sendToActiveMQ(ack, createResponseHandler(command));
449                    acked = true;
450                }
451            }
452
453        } else if (subscriptionId != null) {
454            StompSubscription sub = this.subscriptions.get(subscriptionId);
455            if (sub != null) {
456                MessageAck ack = sub.onStompMessageAck(messageId, activemqTx);
457                if (ack != null) {
458                    sendToActiveMQ(ack, createResponseHandler(command));
459                    acked = true;
460                }
461            }
462        } else {
463            // STOMP v1.0: acking with just a message id is very bogus since the same message id
464            // could have been sent to 2 different subscriptions on the same Stomp connection.
465            // For example, when 2 subs are created on the same topic.
466            for (StompSubscription sub : subscriptionsByConsumerId.values()) {
467                MessageAck ack = sub.onStompMessageAck(messageId, activemqTx);
468                if (ack != null) {
469                    sendToActiveMQ(ack, createResponseHandler(command));
470                    acked = true;
471                    break;
472                }
473            }
474        }
475
476        if (!acked) {
477            throw new ProtocolException("Unexpected ACK received for message-id [" + messageId + "]");
478        }
479    }

481    protected void onStompBegin(StompFrame command) throws ProtocolException {
482        checkConnected();
483
484        Map<String, String> headers = command.getHeaders();
485
486        String stompTx = headers.get(Stomp.Headers.TRANSACTION);
487
488        if (!headers.containsKey(Stomp.Headers.TRANSACTION)) {
489            throw new ProtocolException("Must specify the transaction you are beginning");
490        }
491
492        if (transactions.get(stompTx) != null) {
493            throw new ProtocolException("The transaction was already started: " + stompTx);
494        }
495
496        LocalTransactionId activemqTx = new LocalTransactionId(connectionId, transactionIdGenerator.getNextSequenceId());
497        transactions.put(stompTx, activemqTx);
498
499        TransactionInfo tx = new TransactionInfo();
500        tx.setConnectionId(connectionId);
501        tx.setTransactionId(activemqTx);
502        tx.setType(TransactionInfo.BEGIN);
503
504        sendToActiveMQ(tx, createResponseHandler(command));
505    }

507    protected void onStompCommit(StompFrame command) throws ProtocolException {
508        checkConnected();
509
510        Map<String, String> headers = command.getHeaders();
511
512        String stompTx = headers.get(Stomp.Headers.TRANSACTION);
513        if (stompTx == null) {
514            throw new ProtocolException("Must specify the transaction you are committing");
515        }
516
517        TransactionId activemqTx = transactions.remove(stompTx);
518        if (activemqTx == null) {
519            throw new ProtocolException("Invalid transaction id: " + stompTx);
520        }
521
522        for (StompSubscription sub : subscriptionsByConsumerId.values()) {
523            sub.onStompCommit(activemqTx);
524        }
525
526        TransactionInfo tx = new TransactionInfo();
527        tx.setConnectionId(connectionId);
528        tx.setTransactionId(activemqTx);
529        tx.setType(TransactionInfo.COMMIT_ONE_PHASE);
530
531        sendToActiveMQ(tx, createResponseHandler(command));
532    }

534    protected void onStompAbort(StompFrame command) throws ProtocolException {
535        checkConnected();
536        Map<String, String> headers = command.getHeaders();
537
538        String stompTx = headers.get(Stomp.Headers.TRANSACTION);
539        if (stompTx == null) {
540            throw new ProtocolException("Must specify the transaction you are committing");
541        }
542
543        TransactionId activemqTx = transactions.remove(stompTx);
544        if (activemqTx == null) {
545            throw new ProtocolException("Invalid transaction id: " + stompTx);
546        }
547        for (StompSubscription sub : subscriptionsByConsumerId.values()) {
548            try {
549                sub.onStompAbort(activemqTx);
550            } catch (Exception e) {
551                throw new ProtocolException("Transaction abort failed", false, e);
552            }
553        }
554
555        TransactionInfo tx = new TransactionInfo();
556        tx.setConnectionId(connectionId);
557        tx.setTransactionId(activemqTx);
558        tx.setType(TransactionInfo.ROLLBACK);
559
560        sendToActiveMQ(tx, createResponseHandler(command));
561    }

563    protected void onStompSubscribe(StompFrame command) throws ProtocolException {
564        checkConnected();
565        FrameTranslator translator = findTranslator(command.getHeaders().get(Stomp.Headers.TRANSFORMATION));
566        Map<String, String> headers = command.getHeaders();
567
568        String subscriptionId = headers.get(Stomp.Headers.Subscribe.ID);
569        String destination = headers.get(Stomp.Headers.Subscribe.DESTINATION);
570
571        if (!this.version.equals(Stomp.V1_0) && subscriptionId == null) {
572            throw new ProtocolException("SUBSCRIBE received without a subscription id!");
573        }
574
575        if (destination == null || "".equals(destination)) {
576            throw new ProtocolException("Invalid empty or 'null' Destination header");
577        }
578
579        final ActiveMQDestination actualDest = translator.convertDestination(this, destination, true);
580
581        if (actualDest == null) {
582            throw new ProtocolException("Invalid 'null' Destination.");
583        }
584
585        final ConsumerId id = new ConsumerId(sessionId, consumerIdGenerator.getNextSequenceId());
586        ConsumerInfo consumerInfo = new ConsumerInfo(id);
587        consumerInfo.setPrefetchSize(actualDest.isQueue() ?
588                ActiveMQPrefetchPolicy.DEFAULT_QUEUE_PREFETCH :
589                headers.containsKey("activemq.subscriptionName") ?
590                        ActiveMQPrefetchPolicy.DEFAULT_DURABLE_TOPIC_PREFETCH : ActiveMQPrefetchPolicy.DEFAULT_TOPIC_PREFETCH);
591        consumerInfo.setDispatchAsync(true);
592
593        String browser = headers.get(Stomp.Headers.Subscribe.BROWSER);
594        if (browser != null && browser.equals(Stomp.TRUE)) {
595
596            if (this.version.equals(Stomp.V1_0)) {
597                throw new ProtocolException("Queue Browser feature only valid for Stomp v1.1+ clients!");
598            }
599
600            consumerInfo.setBrowser(true);
601            consumerInfo.setPrefetchSize(ActiveMQPrefetchPolicy.DEFAULT_QUEUE_BROWSER_PREFETCH);
602        }
603
604        String selector = headers.remove(Stomp.Headers.Subscribe.SELECTOR);
605        if (selector != null) {
606            consumerInfo.setSelector("convert_string_expressions:" + selector);
607        }
608
609        IntrospectionSupport.setProperties(consumerInfo, headers, "activemq.");
610
611        if (actualDest.isQueue() && consumerInfo.getSubscriptionName() != null) {
612            throw new ProtocolException("Invalid Subscription: cannot durably subscribe to a Queue destination!");
613        }
614
615        consumerInfo.setDestination(actualDest);
616        consumerInfo.setDispatchAsync(true);
617
618        StompSubscription stompSubscription;
619        if (!consumerInfo.isBrowser()) {
620            stompSubscription = new StompSubscription(this, subscriptionId, consumerInfo, headers.get(Stomp.Headers.TRANSFORMATION), pendingAcksTracker);
621        } else {
622            stompSubscription = new StompQueueBrowserSubscription(this, subscriptionId, consumerInfo, headers.get(Stomp.Headers.TRANSFORMATION), pendingAcksTracker);
623        }
624        stompSubscription.setDestination(actualDest);
625
626        String ackMode = headers.get(Stomp.Headers.Subscribe.ACK_MODE);
627        if (Stomp.Headers.Subscribe.AckModeValues.CLIENT.equals(ackMode)) {
628            stompSubscription.setAckMode(StompSubscription.CLIENT_ACK);
629        } else if (Stomp.Headers.Subscribe.AckModeValues.INDIVIDUAL.equals(ackMode)) {
630            stompSubscription.setAckMode(StompSubscription.INDIVIDUAL_ACK);
631        } else {
632            stompSubscription.setAckMode(StompSubscription.AUTO_ACK);
633        }
634
635        subscriptionsByConsumerId.put(id, stompSubscription);
636        // Stomp v1.0 doesn't need to set this header so we avoid an NPE if not set.
637        if (subscriptionId != null) {
638            subscriptions.put(subscriptionId, stompSubscription);
639        }
640
641        final String receiptId = command.getHeaders().get(Stomp.Headers.RECEIPT_REQUESTED);
642        if (receiptId != null && consumerInfo.getPrefetchSize() > 0) {
643
644            final StompFrame cmd = command;
645            final int prefetch = consumerInfo.getPrefetchSize();
646
647            // Since dispatch could beat the receipt we set prefetch to zero to start and then
648            // once we've sent our Receipt we are safe to turn on dispatch if the response isn't
649            // an error message.
650            consumerInfo.setPrefetchSize(0);
651
652            final ResponseHandler handler = new ResponseHandler() {
653                @Override
654                public void onResponse(ProtocolConverter converter, Response response) throws IOException {
655                    if (response.isException()) {
656                        // Generally a command can fail.. but that does not invalidate the connection.
657                        // We report back the failure but we don't close the connection.
658                        Throwable exception = ((ExceptionResponse)response).getException();
659                        handleException(exception, cmd);
660                    } else {
661                        StompFrame sc = new StompFrame();
662                        sc.setAction(Stomp.Responses.RECEIPT);
663                        sc.setHeaders(new HashMap<String, String>(1));
664                        sc.getHeaders().put(Stomp.Headers.Response.RECEIPT_ID, receiptId);
665                        stompTransport.sendToStomp(sc);
666
667                        ConsumerControl control = new ConsumerControl();
668                        control.setPrefetch(prefetch);
669                        control.setDestination(actualDest);
670                        control.setConsumerId(id);
671
672                        sendToActiveMQ(control, null);
673                    }
674                }
675            };
676
677            sendToActiveMQ(consumerInfo, handler);
678        } else {
679            sendToActiveMQ(consumerInfo, createResponseHandler(command));
680        }
681    }

683    protected void onStompUnsubscribe(StompFrame command) throws ProtocolException {
684        checkConnected();
685        Map<String, String> headers = command.getHeaders();
686
687        ActiveMQDestination destination = null;
688        Object o = headers.get(Stomp.Headers.Unsubscribe.DESTINATION);
689        if (o != null) {
690            destination = findTranslator(command.getHeaders().get(Stomp.Headers.TRANSFORMATION)).convertDestination(this, (String)o, true);
691        }
692
693        String subscriptionId = headers.get(Stomp.Headers.Unsubscribe.ID);
694        if (!this.version.equals(Stomp.V1_0) && subscriptionId == null) {
695            throw new ProtocolException("UNSUBSCRIBE received without a subscription id!");
696        }
697
698        if (subscriptionId == null && destination == null) {
699            throw new ProtocolException("Must specify the subscriptionId or the destination you are unsubscribing from");
700        }
701
702        // check if it is a durable subscription
703        String durable = command.getHeaders().get("activemq.subscriptionName");
704        String clientId = durable;
705        if (!this.version.equals(Stomp.V1_0)) {
706            clientId = connectionInfo.getClientId();
707        }
708
709        if (durable != null) {
710            RemoveSubscriptionInfo info = new RemoveSubscriptionInfo();
711            info.setClientId(clientId);
712            info.setSubscriptionName(durable);
713            info.setConnectionId(connectionId);
714            sendToActiveMQ(info, createResponseHandler(command));
715            return;
716        }
717
718        if (subscriptionId != null) {
719            StompSubscription sub = this.subscriptions.remove(subscriptionId);
720            if (sub != null) {
721                sendToActiveMQ(sub.getConsumerInfo().createRemoveCommand(), createResponseHandler(command));
722                return;
723            }
724        } else {
725            // Unsubscribing using a destination is a bit weird if multiple subscriptions
726            // are created with the same destination.
727            for (Iterator<StompSubscription> iter = subscriptionsByConsumerId.values().iterator(); iter.hasNext();) {
728                StompSubscription sub = iter.next();
729                if (destination != null && destination.equals(sub.getDestination())) {
730                    sendToActiveMQ(sub.getConsumerInfo().createRemoveCommand(), createResponseHandler(command));
731                    iter.remove();
732                    return;
733                }
734            }
735        }
736
737        throw new ProtocolException("No subscription matched.");
738    }

742    protected void onStompConnect(final StompFrame command) throws ProtocolException {
743
744        if (connected.get()) {
745            throw new ProtocolException("Already connected.");
746        }
747
748        final Map<String, String> headers = command.getHeaders();
749
750        // allow anyone to login for now
751        String login = headers.get(Stomp.Headers.Connect.LOGIN);
752        String passcode = headers.get(Stomp.Headers.Connect.PASSCODE);
753        String clientId = headers.get(Stomp.Headers.Connect.CLIENT_ID);
754        String heartBeat = headers.get(Stomp.Headers.Connect.HEART_BEAT);
755        String host = headers.get(Stomp.Headers.Connect.HOST);
756
757        if (heartBeat == null) {
758            heartBeat = defaultHeartBeat;
759        }
760
761        this.version = StompCodec.detectVersion(headers);
762
763        configureInactivityMonitor(heartBeat.trim());
764
765        IntrospectionSupport.setProperties(connectionInfo, headers, "activemq.");
766        connectionInfo.setConnectionId(connectionId);
767        if (clientId != null) {
768            connectionInfo.setClientId(clientId);
769        } else {
770            connectionInfo.setClientId("" + connectionInfo.getConnectionId().toString());
771        }
772        connectionInfo.setClientIp(host);
773
774        connectionInfo.setResponseRequired(true);
775        connectionInfo.setUserName(login);
776        connectionInfo.setPassword(passcode);
777        connectionInfo.setTransportContext(command.getTransportContext());
778
779        sendToActiveMQ(connectionInfo, new ResponseHandler() {
780            @Override
781            public void onResponse(ProtocolConverter converter, Response response) throws IOException {
782
783                if (response.isException()) {
784                    // If the connection attempt fails we close the socket.
785                    Throwable exception = ((ExceptionResponse)response).getException();
786                    handleException(exception, command);
787                    getStompTransport().onException(IOExceptionSupport.create(exception));
788                    return;
789                }
790
791                final SessionInfo sessionInfo = new SessionInfo(sessionId);
792                sendToActiveMQ(sessionInfo, null);
793
794                final ProducerInfo producerInfo = new ProducerInfo(producerId);
795                sendToActiveMQ(producerInfo, new ResponseHandler() {
796                    @Override
797                    public void onResponse(ProtocolConverter converter, Response response) throws IOException {
798
799                        if (response.isException()) {
800                            // If the connection attempt fails we close the socket.
801                            Throwable exception = ((ExceptionResponse)response).getException();
802                            handleException(exception, command);
803                            getStompTransport().onException(IOExceptionSupport.create(exception));
804                        }
805
806                        connected.set(true);
807                        HashMap<String, String> responseHeaders = new HashMap<>();
808
809                        responseHeaders.put(Stomp.Headers.Connected.SESSION, connectionInfo.getClientId());
810                        String requestId = headers.get(Stomp.Headers.Connect.REQUEST_ID);
811                        if (requestId == null) {
812                            // TODO legacy
813                            requestId = headers.get(Stomp.Headers.RECEIPT_REQUESTED);
814                        }
815                        if (requestId != null) {
816                            // TODO legacy
817                            responseHeaders.put(Stomp.Headers.Connected.RESPONSE_ID, requestId);
818                            responseHeaders.put(Stomp.Headers.Response.RECEIPT_ID, requestId);
819                        }
820
821                        responseHeaders.put(Stomp.Headers.Connected.VERSION, version);
822                        responseHeaders.put(Stomp.Headers.Connected.HEART_BEAT,
823                                            String.format("%d,%d", hbWriteInterval, hbReadInterval));
824                        responseHeaders.put(Stomp.Headers.Connected.SERVER, "ActiveMQ/"+BROKER_VERSION);
825
826                        StompFrame sc = new StompFrame();
827                        sc.setAction(Stomp.Responses.CONNECTED);
828                        sc.setHeaders(responseHeaders);
829                        sendToStomp(sc);
830
831                        StompWireFormat format = stompTransport.getWireFormat();
832                        if (format != null) {
833                            format.setStompVersion(version);
834                        }
835                    }
836                });
837            }
838        });
839    }

856    /**
857     * Dispatch a ActiveMQ command
858     *
859     * @param command
860     * @throws IOException
861     */
862    public void onActiveMQCommand(Command command) throws IOException, JMSException {
863        if (command.isResponse()) {
864            Response response = (Response)command;
865            ResponseHandler rh = resposeHandlers.remove(Integer.valueOf(response.getCorrelationId()));
866            if (rh != null) {
867                rh.onResponse(this, response);
868            } else {
869                // Pass down any unexpected errors. Should this close the connection?
870                if (response.isException()) {
871                    Throwable exception = ((ExceptionResponse)response).getException();
872                    handleException(exception, null);
873                }
874            }
875        } else if (command.isMessageDispatch()) {
876            MessageDispatch md = (MessageDispatch)command;
877            StompSubscription sub = subscriptionsByConsumerId.get(md.getConsumerId());
878            if (sub != null) {
879                sub.onMessageDispatch(md);
880            }
881        } else if (command.getDataStructureType() == CommandTypes.KEEP_ALIVE_INFO) {
882            stompTransport.sendToStomp(ping);
883        } else if (command.getDataStructureType() == ConnectionError.DATA_STRUCTURE_TYPE) {
884            // Pass down any unexpected async errors. Should this close the connection?
885            Throwable exception = ((ConnectionError)command).getException();
886            handleException(exception, null);
887        }
888    }

950    protected void configureInactivityMonitor(String heartBeatConfig) throws ProtocolException {
951
952        String[] keepAliveOpts = heartBeatConfig.split(Stomp.COMMA);
953
954        if (keepAliveOpts == null || keepAliveOpts.length != 2) {
955            throw new ProtocolException("Invalid heart-beat header:" + heartBeatConfig, true);
956        } else {
957
958            try {
959                hbReadInterval = (Long.parseLong(keepAliveOpts[0]));
960                hbWriteInterval = Long.parseLong(keepAliveOpts[1]);
961            } catch(NumberFormatException e) {
962                throw new ProtocolException("Invalid heart-beat header:" + heartBeatConfig, true);
963            }
964
965            try {
966                StompInactivityMonitor monitor = this.stompTransport.getInactivityMonitor();
967                monitor.setReadCheckTime((long) (hbReadInterval * hbGracePeriodMultiplier));
968                monitor.setInitialDelayTime(Math.min(hbReadInterval, hbWriteInterval));
969                monitor.setWriteCheckTime(hbWriteInterval);
970                monitor.startMonitoring();
971            } catch(Exception ex) {
972                hbReadInterval = 0;
973                hbWriteInterval = 0;
974            }
975
976            if (LOG.isDebugEnabled()) {
977                LOG.debug("Stomp Connect heartbeat conf RW[{},{}]", hbReadInterval, hbWriteInterval);
978            }
979        }
980    }

994        }

997    /**
998     * Retrieve the STOMP action value from a frame if the value is valid, otherwise
999     * return an unknown string to allow for safe log output.
1000     *
1001     * @param command
1002     *      The STOMP command to fetch an action from.
1003     *
1004     * @return the command action or a safe string to use in logging.
1005     */
1006    protected Object safeGetAction(StompFrame command) {
1007        String result = "<Unknown>";
1008        if (command != null && command.getAction() != null) {
1009            String action = command.getAction().trim();
1010
1011            if (action != null) {
1012                switch (action) {
1013                    case Stomp.Commands.SEND:
1014                    case Stomp.Commands.ACK:
1015                    case Stomp.Commands.NACK:
1016                    case Stomp.Commands.BEGIN:
1017                    case Stomp.Commands.COMMIT:
1018                    case Stomp.Commands.ABORT:
1019                    case Stomp.Commands.SUBSCRIBE:
1020                    case Stomp.Commands.UNSUBSCRIBE:
1021                    case Stomp.Commands.CONNECT:
1022                    case Stomp.Commands.STOMP:
1023                    case Stomp.Commands.DISCONNECT:
1024                        result = action;
1025                        break;
1026                    case Stomp.Commands.SUBSCRIBE_PREFIX:
1027                        result = Stomp.Commands.SUBSCRIBE;
1028                    case Stomp.Commands.UNSUBSCRIBE_PREFIX:
1029                        result = Stomp.Commands.UNSUBSCRIBE;
1030                    default:
1031                        break;
1032                }
1033            }
1034        }
1035
1036        return result;
1037    }
