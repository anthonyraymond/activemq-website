017package org.apache.activemq.command;
018
019import javax.jms.JMSException;
020import javax.jms.TemporaryTopic;
021
022/**
023 * @org.apache.xbean.XBean element="tempTopic" description="An ActiveMQ Temporary Topic Destination"
024 * @openwire:marshaller code="103"
025 * 
026 */
027public class ActiveMQTempTopic extends ActiveMQTempDestination implements TemporaryTopic {
028
029    public static final byte DATA_STRUCTURE_TYPE = CommandTypes.ACTIVEMQ_TEMP_TOPIC;
030    private static final long serialVersionUID = -4325596784597300253L;
031
032    public ActiveMQTempTopic() {
033    }
034
035    public ActiveMQTempTopic(String name) {
036        super(name);
037    }
038
039    public ActiveMQTempTopic(ConnectionId connectionId, long sequenceId) {
040        super(connectionId.getValue(), sequenceId);
041    }
042
043    public byte getDataStructureType() {
044        return DATA_STRUCTURE_TYPE;
045    }
046
047    public boolean isTopic() {
048        return true;
049    }
050
051    public String getTopicName() throws JMSException {
052        return getPhysicalName();
053    }
054
055    public byte getDestinationType() {
056        return TEMP_TOPIC_TYPE;
057    }
058
059    protected String getQualifiedPrefix() {
060        return TEMP_TOPIC_QUALIFED_PREFIX;
061    }
062}
