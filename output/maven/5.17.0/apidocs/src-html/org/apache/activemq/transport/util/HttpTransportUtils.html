017package org.apache.activemq.transport.util;
018
019import javax.servlet.http.HttpServletRequest;
020
021public class HttpTransportUtils {
022
023    public static String generateWsRemoteAddress(HttpServletRequest request) {
024        if (request == null) {
025            throw new IllegalArgumentException("HttpServletRequest must not be null.");
026        }
027
028        StringBuilder remoteAddress = new StringBuilder();
029        String scheme = request.getScheme();
030        remoteAddress.append(scheme != null && scheme.equalsIgnoreCase("https") ? "wss://" : "ws://");
031        remoteAddress.append(request.getRemoteAddr());
032        remoteAddress.append(":");
033        remoteAddress.append(request.getRemotePort());
034        return remoteAddress.toString();
035    }
036
037    public static String generateWsRemoteAddress(HttpServletRequest request, String schemePrefix) {
038        if (request == null) {
039            throw new IllegalArgumentException("HttpServletRequest must not be null.");
040        }
041
042        StringBuilder remoteAddress = new StringBuilder();
043        String scheme = request.getScheme();
044        if (scheme != null && scheme.equalsIgnoreCase("https")) {
045            scheme = schemePrefix + "+wss://";
046        } else {
047            scheme = schemePrefix + "+ws://";
048        }
049
050        remoteAddress.append(scheme);
051        remoteAddress.append(request.getRemoteAddr());
052        remoteAddress.append(":");
053        remoteAddress.append(request.getRemotePort());
054
055        return remoteAddress.toString();
056    }
057}
