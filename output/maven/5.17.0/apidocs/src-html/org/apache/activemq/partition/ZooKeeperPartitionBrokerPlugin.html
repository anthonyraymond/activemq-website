017package org.apache.activemq.partition;
018
019import org.apache.activemq.broker.Broker;
020import org.apache.activemq.broker.BrokerPlugin;
021
022/**
023 * A PartitionBrokerPlugin which gets it's configuration from ZooKeeper.
024 */
025public class ZooKeeperPartitionBrokerPlugin extends PartitionBrokerPlugin {
026
027    String zkAddress = "127.0.0.1:2181";
028    String zkPassword;
029    String zkPath = "/broker-assignments";
030    String zkSessionTmeout = "10s";
031
032    @Override
033    public Broker installPlugin(Broker broker) throws Exception {
034        return new ZooKeeperPartitionBroker(broker, this);
035    }
036
037    public String getZkAddress() {
038        return zkAddress;
039    }
040
041    public void setZkAddress(String zkAddress) {
042        this.zkAddress = zkAddress;
043    }
044
045    public String getZkPassword() {
046        return zkPassword;
047    }
048
049    public void setZkPassword(String zkPassword) {
050        this.zkPassword = zkPassword;
051    }
052
053    public String getZkPath() {
054        return zkPath;
055    }
056
057    public void setZkPath(String zkPath) {
058        this.zkPath = zkPath;
059    }
060
061    public String getZkSessionTmeout() {
062        return zkSessionTmeout;
063    }
064
065    public void setZkSessionTmeout(String zkSessionTmeout) {
066        this.zkSessionTmeout = zkSessionTmeout;
067    }
068}
