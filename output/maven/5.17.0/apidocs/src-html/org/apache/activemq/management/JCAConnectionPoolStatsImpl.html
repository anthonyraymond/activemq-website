017package org.apache.activemq.management;
018
019/**
020 * Statistics for a JCA connection pool
021 * 
022 * 
023 */
024public class JCAConnectionPoolStatsImpl extends JCAConnectionStatsImpl {
025    private CountStatisticImpl closeCount;
026    private CountStatisticImpl createCount;
027    private BoundedRangeStatisticImpl freePoolSize;
028    private BoundedRangeStatisticImpl poolSize;
029    private RangeStatisticImpl waitingThreadCount;
030
031    public JCAConnectionPoolStatsImpl(String connectionFactory, String managedConnectionFactory, TimeStatisticImpl waitTime, TimeStatisticImpl useTime,
032                                      CountStatisticImpl closeCount, CountStatisticImpl createCount, BoundedRangeStatisticImpl freePoolSize, BoundedRangeStatisticImpl poolSize,
033                                      RangeStatisticImpl waitingThreadCount) {
034        super(connectionFactory, managedConnectionFactory, waitTime, useTime);
035        this.closeCount = closeCount;
036        this.createCount = createCount;
037        this.freePoolSize = freePoolSize;
038        this.poolSize = poolSize;
039        this.waitingThreadCount = waitingThreadCount;
040
041        // lets add named stats
042        addStatistic("freePoolSize", freePoolSize);
043        addStatistic("poolSize", poolSize);
044        addStatistic("waitingThreadCount", waitingThreadCount);
045    }
046
047    public CountStatisticImpl getCloseCount() {
048        return closeCount;
049    }
050
051    public CountStatisticImpl getCreateCount() {
052        return createCount;
053    }
054
055    public BoundedRangeStatisticImpl getFreePoolSize() {
056        return freePoolSize;
057    }
058
059    public BoundedRangeStatisticImpl getPoolSize() {
060        return poolSize;
061    }
062
063    public RangeStatisticImpl getWaitingThreadCount() {
064        return waitingThreadCount;
065    }
066
067}
