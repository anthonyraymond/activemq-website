024public interface PerformanceSampler extends Runnable {

030    Long getRampDownTime();

036    void setDuration(long duration);

040    void setInterval(long interval);

044    void setRampUpPercent(long rampUpPercent);

057
058    void finishSampling();
059
060    void sampleData();
061
062    void startSampler(CountDownLatch completionLatch, ClientRunBasis clientRunBasis, long clientRunDuration);
063
064}
