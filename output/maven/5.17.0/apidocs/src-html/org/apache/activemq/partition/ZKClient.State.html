048    public void start() throws Exception {
049        // Grab the lock to make sure that the registration of the ManagedService
050        // won't be updated immediately but that the initial update will happen first
051        synchronized (_lock) {
052            _stateChangeDispatcher.setDaemon(true);
053            _stateChangeDispatcher.start();
054            doStart();
055        }

070    @Override
071    public void close() {
072        if (_stateChangeDispatcher != null) {
073            _stateChangeDispatcher.end();
074            try {
075                _stateChangeDispatcher.join(1000);
076            } catch (Exception e) {
077                LOG.debug("ignored exception", e);
078            }
079        }
080        synchronized(_lock) {
081            if (_zk != null) {
082                try {
083                    changeState(State.NONE);
084                    _zk.close();
085                    Thread th = getSendThread();
086                    if (th != null) {
087                        th.join(1000);
088                    }
089                    _zk = null;
090                } catch (Exception e) {
091                    LOG.debug("ignored exception", e);
092                }
093            }
094        }
095    }

105    protected Object getField(Object obj, String... names) throws Exception {
106        for (String name : names) {
107            obj = getField(obj, name);
108        }
109        return obj;
110    }

112    protected Object getField(Object obj, String name) throws Exception {
113        Class clazz = obj.getClass();
114        while (clazz != null) {
115            for (Field f : clazz.getDeclaredFields()) {
116                if (f.getName().equals(name)) {
117                    f.setAccessible(true);
118                    return f.get(obj);
119                }
120            }
121        }
122        throw new NoSuchFieldError(name);
123    }
124
125    protected void changeState(State newState) {
126        synchronized (_lock) {
127            State oldState = _state;
128            if (oldState != newState) {
129                _stateChangeDispatcher.addEvent(oldState, newState);
130                _state = newState;
131                _lock.notifyAll();
132            }
133        }
134    }

142    protected Object callMethod(Object obj, String name, Object... args) throws Exception {
143        Class clazz = obj.getClass();
144        while (clazz != null) {
145            for (Method m : clazz.getDeclaredMethods()) {
146                if (m.getName().equals(name)) {
147                    m.setAccessible(true);
148                    return m.invoke(obj, args);
149                }
150            }
151        }
152        throw new NoSuchMethodError(name);
153    }

155    protected void tryConnect() {
156        synchronized (_lock) {
157            try {
158                connect();
159            } catch (Throwable e) {
160                LOG.warn("Error while restarting:", e);
161                if (_expiredSessionRecovery == null) {
162                    _expiredSessionRecovery = new ExpiredSessionRecovery();
163                    _expiredSessionRecovery.setDaemon(true);
164                    _expiredSessionRecovery.start();
165                }
166            }
167        }
168    }

170    public void connect() throws UnsupportedEncodingException {
171        synchronized (_lock) {
172            changeState(State.CONNECTING);
173            _zk = _factory.createZooKeeper(this);
174            if (password != null) {
175                _zk.addAuthInfo("digest", ("fabric:" + password).getBytes("UTF-8"));
176            }
177        }
178    }

180    public void process(WatchedEvent event) {
181        if (event.getState() != null) {
182            LOG.debug("event: {}", event.getState());
183            synchronized (_lock) {
184                switch(event.getState()) {
185                    case SyncConnected:
186                        changeState(State.CONNECTED);
187                        break;
188                    case Disconnected:
189                        if (_state != State.NONE) {
190                            changeState(State.RECONNECTING);
191                        }
192                        break;
193                    case Expired:
194                        // when expired, the zookeeper object is invalid and we need to recreate a new one
195                        _zk = null;
196                        LOG.warn("Expiration detected: trying to restart...");
197                        tryConnect();
198                        break;
199                    default:
200                        LOG.warn("Unsupported event state: {}", event.getState());
201                }
202            }
203        }
204    }

206    @Override
207    protected IZooKeeper getZk() {
208        State state = _state;
209        if (state == State.NONE) {
210            throw new IllegalStateException("ZooKeeper client has not been configured yet. You need to either create an ensemble or join one.");
211        } else if (state != State.CONNECTING) {
212            try {
213                waitForConnected();
214            } catch (Exception e) {
215                throw new IllegalStateException("Error waiting for ZooKeeper connection", e);
216            }
217        }
218        IZooKeeper zk = _zk;
219        if (zk == null) {
220            throw new IllegalStateException("No ZooKeeper connection available");
221        }
222        return zk;
223    }

233    public void waitForState(State state, Timespan timeout) throws TimeoutException, InterruptedException {
234        long endTime = (timeout == null ? sessionTimeout : timeout).futureTimeMillis(_clock);
235        if (_state != state) {
236            synchronized (_lock) {
237                while (_state != state) {
238                    ConcurrentUtils.awaitUntil(_clock, _lock, endTime);
239                }
240            }
241        }
242    }

309    private class StateChangeDispatcher extends Thread {
310        private final AtomicBoolean _running = new AtomicBoolean(true);
311        private final BlockingQueue<Boolean> _events = new LinkedBlockingQueue<>();
312
313        private StateChangeDispatcher() {
314            super("ZooKeeper state change dispatcher thread");
315        }
316
317        @Override
318        public void run() {
319            Map<Object, Boolean> history = new IdentityHashMap<>();
320            LOG.info("Starting StateChangeDispatcher");
321            while (_running.get()) {
322                Boolean isConnectedEvent;
323                try {
324                    isConnectedEvent = _events.take();
325                } catch (InterruptedException e) {
326                    continue;
327                }
328                if (!_running.get() || isConnectedEvent == null) {
329                    continue;
330                }
331                Map<Object, Boolean> newHistory = callListeners(history, isConnectedEvent);
332                // we save which event each listener has seen last
333                // we don't update the map in place because we need to get rid of unregistered listeners
334                history = newHistory;
335            }
336            LOG.info("StateChangeDispatcher terminated.");
337        }
338
339        public void end() {
340            _running.set(false);
341            _events.add(false);
342        }
343
344        public void addEvent(ZKClient.State oldState, ZKClient.State newState) {
345            LOG.debug("addEvent: {} => {}", oldState, newState);
346            if (newState == ZKClient.State.CONNECTED) {
347                _events.add(true);
348            } else if (oldState == ZKClient.State.CONNECTED) {
349                _events.add(false);
350            }
351        }
352    }
353
354    protected Map<Object, Boolean> callListeners(Map<Object, Boolean> history, Boolean connectedEvent) {
355        Map<Object, Boolean> newHistory = new IdentityHashMap<>();
356        for (LifecycleListener listener : _listeners) {
357            Boolean previousEvent = history.get(listener);
358            // we propagate the event only if it was not already sent
359            if (previousEvent == null || previousEvent != connectedEvent) {
360                try {
361                    if (connectedEvent) {
362                        listener.onConnected();
363                    } else {
364                        listener.onDisconnected();
365                    }
366                } catch (Throwable e) {
367                    LOG.warn("Exception while executing listener (ignored)", e);
368                }
369            }
370            newHistory.put(listener, connectedEvent);
371        }
372        return newHistory;
373    }

375    private class ExpiredSessionRecovery extends Thread {
376
377        private ExpiredSessionRecovery() {
378            super("ZooKeeper expired session recovery thread");
379        }
380
381        @Override
382        public void run() {
383            LOG.info("Entering recovery mode");
384            synchronized (_lock) {
385                try {
386                    int count = 0;
387                    while (_state == ZKClient.State.NONE) {
388                        try {
389                            count++;
390                            LOG.warn("Recovery mode: trying to reconnect to zookeeper [{}]", count);
391                            ZKClient.this.connect();
392                        } catch (Throwable e) {
393                            LOG.warn("Recovery mode: reconnect attempt failed [{}]... waiting for {}", count, _reconnectTimeout, e);
394                            try {
395                                _lock.wait(_reconnectTimeout.getDurationInMilliseconds());
396                            } catch (InterruptedException e1) {
397                                throw new RuntimeException("Recovery mode: wait interrupted... bailing out", e1);
398                            }
399                        }
400                    }
401                } finally {
402                    _expiredSessionRecovery = null;
403                    LOG.info("Exiting recovery mode.");
404                }
405            }
406        }
407
408    }

426    static private int getPermFromString(String permString) {
427        int perm = 0;
428        for (int i = 0; i < permString.length(); i++) {
429            switch (permString.charAt(i)) {
430                case 'r':
431                    perm |= ZooDefs.Perms.READ;
432                    break;
433                case 'w':
434                    perm |= ZooDefs.Perms.WRITE;
435                    break;
436                case 'c':
437                    perm |= ZooDefs.Perms.CREATE;
438                    break;
439                case 'd':
440                    perm |= ZooDefs.Perms.DELETE;
441                    break;
442                case 'a':
443                    perm |= ZooDefs.Perms.ADMIN;
444                    break;
445                default:
446                    System.err.println("Unknown perm type:" + permString.charAt(i));
447            }
448        }
449        return perm;
450    }

452    private static List<ACL> parseACLs(String aclString) {
453        List<ACL>  acl;
454        String acls[] = aclString.split(",");
455        acl = new ArrayList<>();
456        for (String a : acls) {
457            int firstColon = a.indexOf(':');
458            int lastColon = a.lastIndexOf(':');
459            if (firstColon == -1 || lastColon == -1 || firstColon == lastColon) {
460                System.err.println(a + " does not have the form scheme:id:perm");
461                continue;
462            }
463            ACL newAcl = new ACL();
464            newAcl.setId(new Id(a.substring(0, firstColon), a.substring(firstColon + 1, lastColon)));
465            newAcl.setPerms(getPermFromString(a.substring(lastColon + 1)));
466            acl.add(newAcl);
467        }
468        return acl;
469    }

532    private void doFixACLs(String path, boolean recursive) throws KeeperException, InterruptedException {
533        setACL(path, getNodeACLs(path), -1);
534        if (recursive) {
535            for (String child : getChildren(path)) {
536                doFixACLs(path.equals("/") ? "/" + child : path + "/" + child, recursive);
537            }
538        }
539    }

541    private List<ACL> getNodeACLs(String path) {
542        String acl = doGetNodeACLs(adjustPath(path));
543        if (acl == null) {
544            throw new IllegalStateException("Could not find matching ACLs for " + path);
545        }
546        return parseACLs(acl);
547    }
548
549    protected String doGetNodeACLs(String path) {
550        String longestPath = "";
551        for (String acl : acls.keySet()) {
552            if (acl.length() > longestPath.length() && path.startsWith(acl)) {
553                longestPath = acl;
554            }
555        }
556        return acls.get(longestPath);
557    }
558
559    private void createParents(String path) throws InterruptedException, KeeperException {
560        path = PathUtils.getParentPath(adjustPath(path));
561        path = PathUtils.removeTrailingSlash(path);
562        List<String> paths = new ArrayList<>();
563        while (!path.equals("") && getZk().exists(path, false) == null) {
564            paths.add(path);
565            path = PathUtils.getParentPath(path);
566            path = PathUtils.removeTrailingSlash(path);
567        }
568        Collections.reverse(paths);
569        for (String p : paths) {
570            try {
571                getZk().create(p,
572                        null,
573                        getNodeACLs(p),
574                        CreateMode.PERSISTENT);
575            } catch (KeeperException.NodeExistsException e) {
576                // ok we continue...
577                if (LOG.isDebugEnabled()) {
578                    LOG.debug("parent already exists " + p);
579                }
580            }
581        }
582    }

584    private byte[] toByteData(String data) {
585        if (data == null) {
586            return null;
587        } else {
588            try {
589                return data.getBytes(CHARSET);
590            } catch (UnsupportedEncodingException e) {
591                throw new RuntimeException(e);
592            }
593        }
