017package org.apache.activemq;
018
019import javax.jms.MessageConsumer;
020
021/**
022 * A listener which is notified if a message is available for processing via the
023 * receive methods. Typically on receiving this notification you can call 
024 * {@link MessageConsumer#receiveNoWait()} to get the new message immediately.
025 * 
026 * Note that this notification just indicates a message is available for synchronous consumption,
027 * it does not actually consume the message.
028 * 
029 * 
030 */
031public interface MessageAvailableListener {
032
033    void onMessageAvailable(MessageConsumer consumer);
034
035}
